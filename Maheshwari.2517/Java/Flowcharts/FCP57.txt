import java.util.*;
public class FCP57{
public static void main(String[]args){
Scanner sc =new Scanner(System.in);
int n=sc.nextInt();
int m=sc.nextInt();
int arr[]=new int[n];
for(int i=0 ; i<n ; i++){
arr[i]=sc.nextInt();
}
for( int i=m ; i<n;i++){
arr[i-1]=arr[i];
}
for( int i=0;i<n-1;i++){
System.out.println(arr[i]);
}
}
}
